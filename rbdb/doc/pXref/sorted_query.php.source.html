  65  /** 
  66   * retrieve the data
  67   *
  68   *  This function will retrieve the user data from the database using
  69   * a simple search criterium (i.e. only one field is searched) and
  70   * sorted by a given field.
  71   *
  72   * @param string $db            an open connection to the DBMS
  73   * @param string $where        the field to search for
  74   * @param string $what        what to find
  75   * @param string $how            how to sort the results
  76   *
  77   * @return $result      Query result 
  78   */
  79  function db_query_sort($db, $where, $what, $how)
  80  {
  81      // Set up a query to show what we were requested
  82      if ( ($where == "") || ($what == ""))
  83          // show all
  84      $query = "SELECT * FROM mut_nt";
  85      else if ( $what == "NULL")
  86          // NULL requires a special syntax
  87      $query = "SELECT * FROM mut_nt WHERE $where IS NULL";
  88      else
  89          // find entries CONTAINING requested text
  90          $query = "SELECT * FROM mut_nt WHERE $where like '%$what%'";
  91      
  92      if ($how != "")
  93          $query .= " ORDER BY $how";
  94  
  95  #    echo "<h3>$query</h3>";
  96  
  97      // run the query
  98      if (! ($result = @ mysql_query($query, $db)) )
  99          show_error();
 100  
 101      return $result;
 102  }
