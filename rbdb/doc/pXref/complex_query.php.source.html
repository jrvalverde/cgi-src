  66  /** 
  67   * retrieve the data
  68   *
  69   * Build a complex query and submit it to the database, requesting that
  70   * results be sorted by a given field.
  71   *
  72   * @param resource $db            an open connection to the DBMS
  73   * @param array  $request        the query to run against the database
  74   * @param string $how        how to sort results
  75   * @return mixed $result      Query result     
  76   */
  77  function db_complex_query_sort($db, $request, $how)
  78  {
  79      // Set up a query to show what we were requested
  80      $query = "SELECT * FROM mut_nt";
  81      
  82      $first_cond = 0;
  83      foreach ($request as $cond) {
  84          // check if this condition states something to search
  85          if (($cond['fld'] != "") && ($cond['qry'] != "")) {
  86          // first valid condition is special
  87          if ($first_cond == 0) {
  88          $first_cond = 1;
  89              $query .= " WHERE ";
  90          if ($cond['op'] == "NOT")
  91              // ignore AND and OR on the first valid condition
  92              $query .= " {$cond['op']} ";
  93          }
  94          else
  95              $query .= " {$cond['op']} ";
  96          
  97          if ($cond['qry'] == "NULL")
  98              // NULL requires a special syntax
  99              $query .= " {$cond['fld']} IS NULL ";
 100          else
 101              $query .= " {$cond['fld']} like '%{$cond['qry']}%'";
 102      }
 103      }
 104      if ($how != "")
 105          $query .= " ORDER BY $how";
 106  
 107  #    echo "<h3>$query</h3>";
 108      
 109      // run the query
 110      if (! ($result = @ mysql_query($query, $db)) )
 111          show_error();
 112  
 113      return $result;
 114  }
