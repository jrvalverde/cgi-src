  66  /** 
  67   * Retrieve the data
  68   *
  69   * Build a complex query and submit it to the database, requesting that
  70   * results be sorted by a given field.
  71   *
  72   * @param string $db            an open connection to the DBMS
  73   * @param array  $query        the query issued
  74   * @param string $how       how to sort results
  75   *
  76   * @return $result      Query result     
  77   */
  78  function db_complex_query_sort($db, $request, $how)
  79  {
  80      // Set up a query to show what we were requested
  81      $query = "SELECT * FROM mut_nt";
  82      
  83      $first_cond = 0;
  84      foreach ($request as $cond) {
  85          // check if this condition states something to search
  86          if (($cond['fld'] != "") && ($cond['qry'] != "")) {
  87          // first valid condition is special
  88          if ($first_cond == 0) {
  89          $first_cond = 1;
  90              $query .= " WHERE ";
  91          if ($cond['op'] == "NOT")
  92              // ignore AND and OR on the first valid condition
  93              $query .= " {$cond['op']} ";
  94          }
  95          else
  96              $query .= " {$cond['op']} ";
  97          
  98          if ($cond['qry'] == "NULL")
  99              // NULL requires a special syntax
 100              $query .= " {$cond['fld']} IS NULL ";
 101          else
 102              $query .= " {$cond['fld']} like '%{$cond['qry']}%'";
 103      }
 104      }
 105      if ($how != "")
 106          $query .= " ORDER BY $how";
 107  
 108  #    echo "<h3>$query</h3>";
 109      
 110      // run the query
 111      if (! ($result = @ mysql_query($query, $db)) )
 112          show_error();
 113  
 114      return $result;
 115  }
