  75   * check an e-mail address for validity
  76   * more complex version
  77   */
  78  function check_email($email) {
  79      // checks proper syntax
  80      if(preg_match("/^([a-zA-Z0-9])+([a-zA-Z0-9\._-])*@([a-zA-Z0-9_-])+([a-zA-Z0-9\._-]+)+$/" , $email)) {
  81          // gets domain name
  82          list($username,$domain)=split('@',$email);
  83          // check for its MX records in the DNS and
  84      // attempts a socket connection to mail server
  85      //  NOTE: if the mail server is not available, yet valid,
  86      //  we may reject a valid e-mail address!
  87      $mxhosts = array();
  88          if(!getmxrr($domain, $mxhosts)) {
  89              // no mx records, we have to check domain
  90              if (!fsockopen($domain,25,$errno,$errstr,30)) {
  91                  return false;
  92              } else {
  93                  return true;
  94              }
  95          } else {
  96              // mx records found
  97              foreach ($mxhosts as $host) {
  98                  if (fsockopen($host,25,$errno,$errstr,30)) {
  99                      return true;
 100                  }
 101              }
 102              return false;
 103          }
 104      
 105          if(!fsockopen($domain,25,$errno,$errstr,30)) {
 106              return false;
 107          }
 108          return true;
 109      }
 110      return false;
 111  }
